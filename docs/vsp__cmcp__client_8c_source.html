200 int vsp_cmcp_client_establish_connection(vsp_cmcp_client *cmcp_client)
201 {
202  int ret;
203  struct timespec time_connection_timeout;
204 
205  /* start measuring time for heartbeat timeout */
206  vsp_time_real_timespec_from_now(&time_connection_timeout,
207  VSP_CMCP_NODE_CONNECTION_TIMEOUT);
208 
209  /* enable connecting to server */
210  vsp_cmcp_state_set(cmcp_client->state, VSP_CMCP_CLIENT_TRYING_TO_CONNECT);
211 
212  /* lock state mutex */
213  vsp_cmcp_state_lock(cmcp_client->state);
214  /* wait until connected or waiting timed out */
215  ret = vsp_cmcp_state_await_state(cmcp_client->state,
216  VSP_CMCP_CLIENT_CONNECTED, &time_connection_timeout);
217  /* unlock state mutex */
218  vsp_cmcp_state_unlock(cmcp_client->state);
219 
220  /* check if connected; otherwise set state to disable connecting */
221  VSP_CHECK(ret == 0, vsp_cmcp_state_set(cmcp_client->state,
222  VSP_CMCP_CLIENT_DISCONNECTED); vsp_error_set_num(ENOTCONN); return -1);
223 
224  return 0;
225 }

227 void vsp_cmcp_client_regular_callback(void *param)
228 {
229  vsp_cmcp_client *cmcp_client;
230 
231  /* check parameters; failures are silently ignored */
232  VSP_CHECK(param != NULL, return);
233 
234  cmcp_client = (vsp_cmcp_client*) param;
235 
236  if (vsp_cmcp_state_get(cmcp_client->state) == VSP_CMCP_CLIENT_CONNECTED
237  && vsp_time_real_timespec_passed(
238  &cmcp_client->time_connection_timeout) == 0) {
239  /* connection to server timed out */
240  vsp_cmcp_state_set(cmcp_client->state,
241  VSP_CMCP_CLIENT_DISCONNECTED);
242  /* connection establishment will not be automatically retried */
243  /* inform about lost connection by invoking callback function */
244  if (cmcp_client->disconnect_cb != NULL) {
245  /* callback function registered; invoke it */
246  cmcp_client->disconnect_cb(cmcp_client->callback_param);
247  }
248  }

251 void vsp_cmcp_client_message_callback(void *param,
252  vsp_cmcp_message *cmcp_message)
253 {
254  vsp_cmcp_client *cmcp_client;
255  vsp_cmcp_datalist *cmcp_datalist;
256  uint16_t topic_id, sender_id, command_id;
257  int state;
258 
259  /* check parameters; failures are silently ignored */
260  VSP_CHECK(param != NULL && cmcp_message != NULL, return);
261 
262  cmcp_client = (vsp_cmcp_client*) param;
263 
264  /* get message IDs */
265  topic_id = vsp_cmcp_message_get_topic_id(cmcp_message);
266  sender_id = vsp_cmcp_message_get_sender_id(cmcp_message);
267  command_id = vsp_cmcp_message_get_command_id(cmcp_message);
268 
269  /* only receive server messages */
270  VSP_CHECK((sender_id & 1) == 0, return);
271 
272  /* get data list */
273  cmcp_datalist = vsp_cmcp_message_get_datalist(cmcp_message);
274  /* check data list; failures are silently ignored */
275  VSP_CHECK(cmcp_datalist != NULL, return);
276 
277  /* get current state */
278  state = vsp_cmcp_state_get(cmcp_client->state);
279 
280  /* check if message is from connected server */
281  if (state == VSP_CMCP_CLIENT_CONNECTED
282  && sender_id == cmcp_client->server_id) {
283  /* reset timeout time */
284  vsp_time_real_timespec_from_now(&cmcp_client->time_connection_timeout,
285  VSP_CMCP_NODE_CONNECTION_TIMEOUT);
286  }
287 
288  /* check if internal control message received */
289  if (vsp_cmcp_message_get_type(cmcp_message)
290  == VSP_CMCP_MESSAGE_TYPE_CONTROL) {
291  /* handle control message */
292  vsp_cmcp_client_handle_control_message(cmcp_client, sender_id,
293  command_id, cmcp_datalist);
294  } else {
295  /* check if message is broadcasted or directed to this node */
296  VSP_CHECK(topic_id == VSP_CMCP_SERVER_BROADCAST_TOPIC_ID
297  || topic_id == cmcp_client->id, return);
298  /* handle data message */
299  if (cmcp_client->message_cb != NULL) {
300  /* callback function registered; invoke it */
301  cmcp_client->message_cb(cmcp_client->callback_param, command_id,
302  cmcp_datalist);
303  }
304  }
305 }

307 void vsp_cmcp_client_handle_control_message(vsp_cmcp_client *cmcp_client,
308  uint16_t sender_id, uint16_t command_id, vsp_cmcp_datalist *cmcp_datalist)
309 {
310  int ret, state;
311  uint64_t *client_nonce;
312 
313  /* get current state */
314  state = vsp_cmcp_state_get(cmcp_client->state);
315 
316  /* process control message */
317  if (state == VSP_CMCP_CLIENT_TRYING_TO_CONNECT) {
318  /* check if server heartbeat received */
319  VSP_CHECK(command_id == VSP_CMCP_COMMAND_SERVER_HEARTBEAT, return);
320  /* store server node ID */
321  cmcp_client->server_id = sender_id;
322  /* server heartbeat received */
323  vsp_cmcp_state_set(cmcp_client->state,
324  VSP_CMCP_CLIENT_HEARTBEAT_RECEIVED);
325  /* send announcement message */
326  ret = vsp_cmcp_client_send_announcement(cmcp_client);
327  /* check for errors; failures are silently ignored */
328  VSP_CHECK(ret == 0, return);
329  } else if (state == VSP_CMCP_CLIENT_HEARTBEAT_RECEIVED) {
330  /* check if (negative) acknowledge received from connected server */
331  VSP_CHECK(sender_id == cmcp_client->server_id
332  && (command_id == VSP_CMCP_COMMAND_SERVER_ACK_CLIENT
333  || command_id == VSP_CMCP_COMMAND_SERVER_NACK_CLIENT), return);
334  /* get client nonce */
335  client_nonce = vsp_cmcp_datalist_get_data_item(cmcp_datalist,
336  VSP_CMCP_PARAMETER_NONCE, sizeof(uint64_t));
337  /* check data list item (nonce); failures are silently ignored */
338  VSP_CHECK(client_nonce != NULL, return);
339  /* ignore message if nonce does not match the nonce of this node */
340  VSP_CHECK(*client_nonce == cmcp_client->nonce, return);
341 
342  if (command_id == VSP_CMCP_COMMAND_SERVER_ACK_CLIENT) {
343  /* acknowledge received, connected */
344  vsp_cmcp_state_set(cmcp_client->state, VSP_CMCP_CLIENT_CONNECTED);
345  /* initialize timeout time */
346  vsp_time_real_timespec_from_now(
347  &cmcp_client->time_connection_timeout,
348  VSP_CMCP_NODE_CONNECTION_TIMEOUT);
349  } else if (command_id == VSP_CMCP_COMMAND_SERVER_NACK_CLIENT) {
350  /* negative acknowledge received, rejected */
351  vsp_cmcp_state_set(cmcp_client->state,
352  VSP_CMCP_CLIENT_TRYING_TO_CONNECT);
353  /* client ID is already registered to server; regenerate node ID */
354  vsp_cmcp_node_generate_id(cmcp_client->cmcp_node);
355  cmcp_client->id = vsp_cmcp_node_get_id(cmcp_client->cmcp_node);
356  /* connection establishment will be automatically retried when next
357  * server heartbeat signal is received */
358  }
359  }
360 }

362 int vsp_cmcp_client_send_announcement(vsp_cmcp_client *cmcp_client)
363 {
364  int ret;
365  int success;
366  vsp_cmcp_datalist *cmcp_datalist;
367 
368  /* initialize local variables */
369  success = 0;
370  cmcp_datalist = NULL;
371 
372  /* create identification nonce */
373  cmcp_client->nonce = vsp_random_get();
374 
375  /* create announcement command message */
376  cmcp_datalist = vsp_cmcp_datalist_create();
377  /* check for errors */
378  VSP_ASSERT(cmcp_datalist != NULL);
379  /* add nonce parameter as a data list item */
380  ret = vsp_cmcp_datalist_add_item(cmcp_datalist, VSP_CMCP_PARAMETER_NONCE,
381  sizeof(uint64_t), &cmcp_client->nonce);
382  /* vsp_error_num() is set by vsp_cmcp_datalist_add_item() */
383  VSP_CHECK(ret == 0, goto error_exit);
384  /* send message */
385  ret = vsp_cmcp_node_create_send_message(cmcp_client->cmcp_node,
386  VSP_CMCP_MESSAGE_TYPE_CONTROL, cmcp_client->server_id, cmcp_client->id,
387  VSP_CMCP_COMMAND_CLIENT_ANNOUNCE, cmcp_datalist);
388  /* vsp_error_num() is set by vsp_cmcp_node_create_send_message() */
389  VSP_CHECK(ret == 0, goto error_exit);
390 
391  /* success: clean up and exit */
392  goto cleanup_exit;
393 
394  error_exit:
395  /* vsp_error_num() is already set */
396  success = -1;
397  /* fall through to cleanup_exit */
398 
399  cleanup_exit:
400  /* clean up */
401  if (cmcp_datalist != NULL) {
402  vsp_cmcp_datalist_free(cmcp_datalist);
403  cmcp_datalist = NULL;
404  }
405  return success;
406 }
