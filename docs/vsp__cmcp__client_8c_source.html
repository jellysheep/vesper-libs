103 int vsp_cmcp_client_establish_connection(vsp_cmcp_client *cmcp_client)
104 {
105  int ret;
106  int data_length;
107  void *message_buffer;
108  vsp_cmcp_message *cmcp_message;
109  double time_now, time_connection_timeout;
110  uint16_t topic_id, sender_id, command_id;
111 
112  /* check parameter */
113  VSP_CHECK(cmcp_client != NULL, vsp_error_set_num(EINVAL); return -1);
114 
115  /* initialize message data */
116  message_buffer = NULL;
117  cmcp_message = NULL;
118 
119  /* initialize IDs */
120  topic_id = 0;
121  sender_id = 0;
122  command_id = 0;
123 
124  /* start measuring time for heartbeat timeout */
125  time_now = vsp_time_real();
126  time_connection_timeout =
127  time_now + (VSP_CMCP_CLIENT_CONNECTION_TIMEOUT / 1000.0);
128 
129  do {
130  /* try to receive message */
131  data_length = vsp_cmcp_node_recv_message(cmcp_client->cmcp_node,
132  &message_buffer);
133  /* measure passed time */
134  time_now = vsp_time_real();
135  /* check error: in case of failure just retry */
136  VSP_CHECK(data_length > 0, goto retry);
137  /* parse message data */
138  cmcp_message = vsp_cmcp_message_create_parse(data_length,
139  message_buffer);
140  /* check error: in case of failure clean up and retry */
141  VSP_CHECK(cmcp_message != NULL, nn_freemsg(message_buffer); goto retry);
142  /* get and check message data; in case of failure just retry */
143  ret = vsp_cmcp_message_get_id(cmcp_message,
144  VSP_CMCP_MESSAGE_TOPIC_ID, &topic_id);
145  VSP_ASSERT(ret == 0, goto cleanup_retry);
146  ret = vsp_cmcp_message_get_id(cmcp_message,
147  VSP_CMCP_MESSAGE_SENDER_ID, &sender_id);
148  VSP_ASSERT(ret == 0, goto cleanup_retry);
149  ret = vsp_cmcp_message_get_id(cmcp_message,
150  VSP_CMCP_MESSAGE_COMMAND_ID, &command_id);
151  VSP_ASSERT(ret == 0, goto cleanup_retry);
152  /* check if server heartbeat received, else retry */
153  VSP_CHECK(topic_id == VSP_CMCP_BROADCAST_TOPIC_ID
154  && sender_id != 0 && (sender_id & 1) == 0
155  && command_id == VSP_CMCP_COMMAND_HEARTBEAT, goto cleanup_retry);
156  /* success, do not retry */
157  break;
158  /* failure: retry */
159  cleanup_retry:
160  /* clean up */
161  ret = vsp_cmcp_message_free(cmcp_message);
162  VSP_ASSERT(ret == 0,
163  /* failures are silently ignored in release build */);
164  cmcp_message = NULL;
165  ret = nn_freemsg(message_buffer);
166  VSP_ASSERT(ret == 0,
167  /* failures are silently ignored in release build */);
168  message_buffer = NULL;
169  retry:
170  continue;
171  } while (time_now < time_connection_timeout);
172 
173  /* check for error */
174  VSP_CHECK(cmcp_message != NULL, vsp_error_set_num(ENOTCONN); return -1);
175 
176  /* server heartbeat received */
177 
178  /* clean up */
179  ret = vsp_cmcp_message_free(cmcp_message);
180  VSP_ASSERT(ret == 0,
181  /* failures are silently ignored in release build */);
182  cmcp_message = NULL;
183  ret = nn_freemsg(message_buffer);
184  VSP_ASSERT(ret == 0,
185  /* failures are silently ignored in release build */);
186  message_buffer = NULL;
187 
188  /* success */
189  return 0;
190 }

192 void *vsp_cmcp_client_run(void *param)
193 {
194  vsp_cmcp_client *cmcp_client;
195 
196  /* check parameter */
197  VSP_ASSERT(param != NULL, vsp_error_set_num(EINVAL); return (void*) -1);
198 
199  cmcp_client = (vsp_cmcp_client*) param;
200 
201  /* check sockets are initialized and thread was started */
202  VSP_ASSERT(cmcp_client->cmcp_node->state == VSP_CMCP_NODE_STARTING,
203  vsp_error_set_num(EINVAL); return (void*) -1);
204 
205  /* lock mutex */
206  pthread_mutex_lock(&cmcp_client->cmcp_node->mutex);
207  /* mark thread as running */
208  cmcp_client->cmcp_node->state = VSP_CMCP_NODE_RUNNING;
209  /* signal waiting thread */
210  pthread_cond_signal(&cmcp_client->cmcp_node->condition);
211  /* unlock mutex */
212  pthread_mutex_unlock(&cmcp_client->cmcp_node->mutex);
213 
214  /* reception loop */
215  while (cmcp_client->cmcp_node->state == VSP_CMCP_NODE_RUNNING) {
216 
217  }
218 
219  /* check thread was requested to stop */
220  VSP_ASSERT(cmcp_client->cmcp_node->state == VSP_CMCP_NODE_STOPPING,
221  vsp_error_set_num(EINTR); return (void*) -1);
222 
223  /* signal that thread has stopped */
224  cmcp_client->cmcp_node->state = VSP_CMCP_NODE_INITIALIZED;
225 
226  /* success */
227  return (void*) 0;
228 }
