91  /* check parameters */

100  cmcp_client->subscribe_socket = nn_socket(AF_SP, NN_SUB);
101 
102  /* connect sockets */

104  /* check error set by nanomsg */
105  VSP_ASSERT(ret >= 0, return -1);
106 
107  ret = nn_connect(cmcp_client->subscribe_socket, subscribe_address);
108  /* check error set by nanomsg */
109  VSP_ASSERT(ret >= 0, return -1);
110 
111  /* set state */
112  cmcp_client->state = VSP_CMCP_CLIENT_INITIALIZED;
113  /* sockets successfully connected */

172 int vsp_cmcp_client_stop(vsp_cmcp_client *cmcp_client)
173 {
174  int ret;
175 
176  /* check parameter */
177  VSP_ASSERT(cmcp_client != NULL, vsp_error_set_num(EINVAL); return -1);
178 
179  /* wait until thread has started */
180  while (cmcp_client->state == VSP_CMCP_CLIENT_STARTING) {
181  /* do nothing */
182  }
183 
184  /* check thread is running */
185  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_RUNNING,
186  vsp_error_set_num(EINVAL); return -1);
187 
188  /* stop reception thread */
189  cmcp_client->state = VSP_CMCP_CLIENT_STOPPING;
190 
191  /* wait for thread to join */
192  ret = pthread_join(cmcp_client->thread, NULL);
193 
194  /* check thread has successfully stopped */
195  VSP_ASSERT(ret == 0, vsp_error_set_num(EINTR); return -1);
196 
197  /* check state */
198  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_INITIALIZED,
199  vsp_error_set_num(EINTR); return -1);
200 
201  /* success */
202  return 0;
203 }

223 
224  }
225 
226  /* check thread was requested to stop */
227  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_STOPPING,
228  vsp_error_set_num(EINTR); return (void*) -1);
