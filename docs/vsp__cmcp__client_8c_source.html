80 int vsp_cmcp_client_connect(vsp_cmcp_client *cmcp_client,
81  const char *publish_address, const char *subscribe_address)
82 {
83  int ret;
84 
85  /* check parameters */
86  VSP_ASSERT(cmcp_client != NULL && publish_address != NULL
87  && subscribe_address != NULL, vsp_error_set_num(EINVAL); return -1);
88  /* check sockets not yet initialized */
89  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_UNINITIALIZED,
90  vsp_error_set_num(EALREADY); return -1);
91 
92  /* initialize sockets */
93  cmcp_client->publish_socket = nn_socket(AF_SP, NN_PUB);
94  cmcp_client->subscribe_socket = nn_socket(AF_SP, NN_SUB);
95 
96  /* connect sockets */
97  ret = nn_connect(cmcp_client->publish_socket, publish_address);
98  /* check error set by nanomsg */
99  VSP_ASSERT(ret >= 0, return -1);
100 
101  ret = nn_connect(cmcp_client->subscribe_socket, subscribe_address);
102  /* check error set by nanomsg */
103  VSP_ASSERT(ret >= 0, return -1);
104 
105  /* set state */
106  cmcp_client->state = VSP_CMCP_CLIENT_INITIALIZED;
107  /* sockets successfully connected */
108  return 0;
109 }

122  /* disconnect sockets */

124  /* check error set by nanomsg */

127  ret = nn_close(cmcp_client->subscribe_socket);
128  /* check error set by nanomsg */
129  VSP_ASSERT(ret == 0, return -1);
130 
131  /* deinitialize sockets */

133  cmcp_client->subscribe_socket = -1;
134 
135  /* set state */
