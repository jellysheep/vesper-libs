68 vsp_cmcp_client *vsp_cmcp_client_create(void)
69 {
70  vsp_cmcp_client *cmcp_client;
71  /* allocate memory */
72  VSP_ALLOC(cmcp_client, vsp_cmcp_client, return NULL);
73  /* initialize struct data */
74  cmcp_client->state = VSP_CMCP_CLIENT_UNINITIALIZED;
75  /* generate client ID that is odd (first bit set)
76  * but does not equal broadcast topic ID */
77  do {
78  cmcp_client->id = (uint16_t) (vsp_random_get() | 1);
79  } while (cmcp_client->id == VSP_CMCP_BROADCAST_TOPIC_ID);
80  cmcp_client->publish_socket = -1;
81  cmcp_client->subscribe_socket = -1;
82  pthread_mutex_init(&cmcp_client->mutex, NULL);
83  pthread_cond_init(&cmcp_client->condition, NULL);
84  /* return struct pointer */
85  return cmcp_client;
86 }

89 {
90  int ret;
91  int success;
92 
93  success = 0;
94 
95  /* check parameter */
96  VSP_ASSERT(cmcp_client != NULL, vsp_error_set_num(EINVAL); return -1);
97 
98  if (cmcp_client->state > VSP_CMCP_CLIENT_INITIALIZED) {
99  /* worker thread is running, stop it */
100  ret = vsp_cmcp_client_stop(cmcp_client);
101  /* check for error */
102  VSP_ASSERT(ret == 0, success = -1);
103  }
104 
105  if (cmcp_client->state > VSP_CMCP_CLIENT_UNINITIALIZED) {
106  /* close publish socket */
107  ret = nn_close(cmcp_client->publish_socket);
108  /* check error set by nanomsg */
109  VSP_ASSERT(ret == 0, success = -1);
110 
111  /* close subscribe socket */
112  ret = nn_close(cmcp_client->subscribe_socket);
113  /* check error set by nanomsg */
114  VSP_ASSERT(ret == 0, success = -1);
115  }
116  pthread_mutex_destroy(&cmcp_client->mutex);
117  pthread_cond_destroy(&cmcp_client->condition);
118  /* free memory */
119  VSP_FREE(cmcp_client);

170 int vsp_cmcp_client_start(vsp_cmcp_client *cmcp_client)
171 {
172  int ret;
173 
174  /* check parameter */
175  VSP_ASSERT(cmcp_client != NULL, vsp_error_set_num(EINVAL); return -1);
176 
177  /* check sockets are initialized and thread is not running */
178  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_INITIALIZED,
179  vsp_error_set_num(EINVAL); return -1);
180 
181  /* lock state mutex */
182  pthread_mutex_lock(&cmcp_client->mutex);
183 
184  /* mark thread as starting */
185  cmcp_client->state = VSP_CMCP_CLIENT_STARTING;
186 
187  /* start reception thread */
188  ret = pthread_create(&cmcp_client->thread, NULL, vsp_cmcp_client_run,
189  (void*) cmcp_client);
190 
191  /* check for pthread errors */
192  VSP_ASSERT(ret == 0, pthread_mutex_unlock(&cmcp_client->mutex);
193  vsp_error_set_num(ret); return -1);
194 
195  /* wait until thread is running */
196  ret = pthread_cond_wait(&cmcp_client->condition, &cmcp_client->mutex);
197  /* unlock mutex again */
198  pthread_mutex_unlock(&cmcp_client->mutex);
199  /* check for condition waiting errors */
200  VSP_ASSERT(ret == 0, vsp_error_set_num(ret); return -1);
201 
202  /* check thread is running */
203  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_RUNNING,
204  vsp_error_set_num(EINVAL); return -1);
205 
206  /* success */
207  return 0;
208 }

239 {
240  vsp_cmcp_client *cmcp_client;
241 
242  /* check parameter */
243  VSP_ASSERT(param != NULL, vsp_error_set_num(EINVAL); return (void*) -1);
244

246

254  cmcp_client->state = VSP_CMCP_CLIENT_RUNNING;
255  /* signal waiting thread */
256  pthread_cond_signal(&cmcp_client->condition);
257  /* unlock mutex */
258  pthread_mutex_unlock(&cmcp_client->mutex);
259 
260  /* reception loop */
261  while (cmcp_client->state == VSP_CMCP_CLIENT_RUNNING) {
262 
263  }
264 
265  /* check thread was requested to stop */
266  VSP_ASSERT(cmcp_client->state == VSP_CMCP_CLIENT_STOPPING,
267  vsp_error_set_num(EINTR); return (void*) -1);
268 
269  /* signal that thread has stopped */
270  cmcp_client->state = VSP_CMCP_CLIENT_INITIALIZED;
271 
272  /* success */
