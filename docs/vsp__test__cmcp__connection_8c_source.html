76 MU_TEST(vsp_test_cmcp_connection_test)
77 {
78  int ret;
79  /* bind */
80  ret = vsp_cmcp_server_bind(global_cmcp_server, SERVER_PUBLISH_ADDRESS,
81  SERVER_SUBSCRIBE_ADDRESS);
82  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
83  /* connect */
84  ret = vsp_cmcp_client_connect(global_cmcp_client, SERVER_SUBSCRIBE_ADDRESS,
85  SERVER_PUBLISH_ADDRESS);
86  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
87  /* start server */
88  ret = vsp_cmcp_server_start(global_cmcp_server);
89  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
90  /* start client */
91  ret = vsp_cmcp_client_start(global_cmcp_client);
92  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
93  /* stop client */
94  ret = vsp_cmcp_client_stop(global_cmcp_client);
95  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
96  /* stop server */
97  ret = vsp_cmcp_server_stop(global_cmcp_server);
98  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
99  /* disconnect */
100  ret = vsp_cmcp_client_disconnect(global_cmcp_client);
101  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
102  /* unbind */
103  ret = vsp_cmcp_server_unbind(global_cmcp_server);
104  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
105 }
