93 int vsp_test_cmcp_announcement_cb(void *callback_param, uint16_t client_id)
94 {
95  /* check if callback parameter equals global server object */
96  mu_assert_abort(callback_param == global_cmcp_server,
97  vsp_error_str(EINVAL));
98  /* check if client ID is valid */
99  mu_assert_abort(client_id != VSP_CMCP_SERVER_BROADCAST_TOPIC_ID
100  && client_id != VSP_CMCP_CLIENT_BROADCAST_TOPIC_ID
101  && (client_id & 1) == 1, vsp_error_str(EINVAL));
102  /* check if test state is correct */
103  mu_assert_abort(vsp_cmcp_state_get(global_test_state)
104  == VSP_TEST_CMCP_NOT_CONNECTED, vsp_error_str(EINVAL));
105  /* store client ID */
106  global_cmcp_client_id = client_id;
107  /* update test state */
108  vsp_cmcp_state_set(global_test_state, VSP_TEST_CMCP_CONNECTED);
109  /* accept new client */
110  return 0;
111 }

337 MU_TEST(vsp_test_cmcp_communication_test)
338 {
339  int ret;
340  vsp_cmcp_datalist *cmcp_datalist;
341  struct timespec time_test_timeout;
342 
343  /* check if test state is correct */
344  mu_assert_abort(vsp_cmcp_state_get(global_test_state)
345  == VSP_TEST_CMCP_CONNECTED, vsp_error_str(EINVAL));
346 
347  /* create data list */
348  cmcp_datalist = vsp_cmcp_datalist_create();
349  mu_assert_abort(cmcp_datalist != NULL, vsp_error_str(vsp_error_num()));
350  /* add a data list item */
351  ret = vsp_cmcp_datalist_add_item(cmcp_datalist, VSP_TEST_DATALIST_ITEM1_ID,
352  VSP_TEST_DATALIST_ITEM1_LENGTH, VSP_TEST_DATALIST_ITEM1_DATA);
353  mu_assert_abort(ret == 0, vsp_error_str(vsp_error_num()));
354  /* send server message */
355  ret = vsp_cmcp_server_send(global_cmcp_server, global_cmcp_client_id,
356  VSP_TEST_MESSAGE_COMMAND_ID, cmcp_datalist);
357  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
358  /* free data list */
359  vsp_cmcp_datalist_free(cmcp_datalist);
360 
361  /* start measuring time for test timeout */
362  vsp_time_real_timespec_from_now(&time_test_timeout,
363  VSP_TEST_CMCP_TIMEOUT);
364 
365  /* lock state mutex */
366  vsp_cmcp_state_lock(global_test_state);
367  /* wait until client message received or waiting timed out */
368  ret = vsp_cmcp_state_await_state(global_test_state,
369  VSP_TEST_CMCP_SERVER_MESSAGE_RECEIVED, &time_test_timeout);
370  /* unlock state mutex */
371  vsp_cmcp_state_unlock(global_test_state);
372  /* check if test was successful */
373  mu_assert(ret == 0, vsp_error_str(ETIMEDOUT));
374 
375  /* create data list */
376  cmcp_datalist = vsp_cmcp_datalist_create();
377  mu_assert_abort(cmcp_datalist != NULL, vsp_error_str(vsp_error_num()));
378  /* add a data list item */
379  ret = vsp_cmcp_datalist_add_item(cmcp_datalist, VSP_TEST_DATALIST_ITEM2_ID,
380  VSP_TEST_DATALIST_ITEM2_LENGTH, VSP_TEST_DATALIST_ITEM2_DATA);
381  mu_assert_abort(ret == 0, vsp_error_str(vsp_error_num()));
382  /* send server message */
383  ret = vsp_cmcp_client_send(global_cmcp_client, VSP_TEST_MESSAGE_COMMAND_ID,
384  cmcp_datalist);
385  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
386  /* free data list */
387  vsp_cmcp_datalist_free(cmcp_datalist);
388 
389  /* disconnect client by freeing it */
390  vsp_cmcp_client_free(global_cmcp_client);
391  global_cmcp_client = NULL;
392 
393  /* start measuring time for test timeout */
394  vsp_time_real_timespec_from_now(&time_test_timeout,
395  VSP_TEST_CMCP_TIMEOUT);
396 
397  /* lock state mutex */
398  vsp_cmcp_state_lock(global_test_state);
399  /* wait until test completed or waiting timed out */
400  ret = vsp_cmcp_state_await_state(global_test_state,
401  VSP_TEST_CMCP_DISCONNECTED, &time_test_timeout);
402  /* unlock state mutex */
403  vsp_cmcp_state_unlock(global_test_state);
404  /* check if test was successful */
405  mu_assert(ret == 0, vsp_error_str(ETIMEDOUT));
406 }
