100 int vsp_cmcp_state_wait(vsp_cmcp_state *cmcp_state,
101  struct timespec *timeout_time)
102 {
103  int ret;
104  int success;
105 
106  success = 0;
107  if (timeout_time != NULL) {
108  /* wait until thread is running, with timeout */
109  ret = pthread_cond_timedwait(&cmcp_state->condition,
110  &cmcp_state->mutex, timeout_time);
111  if (ret == ETIMEDOUT) {
112  vsp_error_set_num(ETIMEDOUT);
113  success = -1;
114  } else {
115  VSP_ASSERT(ret == 0);
116  }
117  } else {
118  /* wait until thread is running, without timeout */
119  ret = pthread_cond_wait(&cmcp_state->condition, &cmcp_state->mutex);
120  VSP_ASSERT(ret == 0);
121  }
122  /* success */
123  return success;
124 }
