36 vsp_cmcp_node *vsp_cmcp_node_create(vsp_cmcp_node_type node_type)
37 {
38  vsp_cmcp_node *cmcp_node;
39  /* check parameter */
40  VSP_CHECK(node_type == VSP_CMCP_NODE_SERVER
41  || node_type == VSP_CMCP_NODE_CLIENT,
42  vsp_error_set_num(EINVAL); return NULL);
43  /* allocate memory */
44  VSP_ALLOC(cmcp_node, vsp_cmcp_node, return NULL);
45  /* initialize struct data */
46  cmcp_node->node_type = node_type;
47  cmcp_node->state = VSP_CMCP_NODE_UNINITIALIZED;
48  /* generate node ID that does not equal broadcast topic ID */
49  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
50  /* node type: ID is even (first bit cleared) */
51  do {
52  cmcp_node->id = (uint16_t) (vsp_random_get() << 1);
53  } while (cmcp_node->id == VSP_CMCP_BROADCAST_TOPIC_ID);
54  } else {
55  /* node type: ID is odd (first bit set) */
56  do {
57  cmcp_node->id = (uint16_t) (vsp_random_get() | 1);
58  } while (cmcp_node->id == VSP_CMCP_BROADCAST_TOPIC_ID);
59  }
60  cmcp_node->publish_socket = -1;
61  cmcp_node->subscribe_socket = -1;
62  pthread_mutex_init(&cmcp_node->mutex, NULL);
63  pthread_cond_init(&cmcp_node->condition, NULL);
64  /* return struct pointer */
65  return cmcp_node;
66 }

68 int vsp_cmcp_node_free(vsp_cmcp_node *cmcp_node)
69 {
70  int ret;
71  int success;
72 
73  /* check parameter */
74  VSP_CHECK(cmcp_node != NULL, vsp_error_set_num(EINVAL); return -1);
75 
76  success = 0;
77 
78  /* clean up socket connections */
79  if (cmcp_node->state > VSP_CMCP_NODE_INITIALIZED) {
80  /* worker thread is running, stop it */
81  ret = vsp_cmcp_node_stop(cmcp_node);
82  /* check for error */
83  VSP_ASSERT(ret == 0, success = -1);
84  }
85 
86  if (cmcp_node->state > VSP_CMCP_NODE_UNINITIALIZED) {
87  /* close publish socket */
88  ret = nn_close(cmcp_node->publish_socket);
89  /* check error set by nanomsg */
90  VSP_ASSERT(ret == 0, success = -1);
91 
92  /* close subscribe socket */
93  ret = nn_close(cmcp_node->subscribe_socket);
94  /* check error set by nanomsg */
95  VSP_ASSERT(ret == 0, success = -1);
96  }
97 
98  /* clean up mutex and condition variable */
99  pthread_mutex_destroy(&cmcp_node->mutex);
100  pthread_cond_destroy(&cmcp_node->condition);
101 
102  /* free memory */
103  VSP_FREE(cmcp_node);
104  return success;
105 }

107 int vsp_cmcp_node_connect(vsp_cmcp_node *cmcp_node,
108  const char *publish_address, const char *subscribe_address)
109 {
110  int ret;
111 
112  /* check parameters */
113  VSP_CHECK(cmcp_node != NULL && publish_address != NULL
114  && subscribe_address != NULL, vsp_error_set_num(EINVAL); return -1);
115  /* check sockets not yet initialized */
116  VSP_CHECK(cmcp_node->state == VSP_CMCP_NODE_UNINITIALIZED,
117  vsp_error_set_num(EALREADY); return -1);
118 
119  /* initialize and connect publish socket */
120  cmcp_node->publish_socket = nn_socket(AF_SP, NN_PUB);
121  /* check error set by nanomsg */
122  VSP_CHECK(cmcp_node->publish_socket != -1, return -1);
123  /* connect or bind socket */
124  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
125  ret = nn_bind(cmcp_node->publish_socket, publish_address);
126  } else {
127  ret = nn_connect(cmcp_node->publish_socket, publish_address);
128  }
129  /* check error set by nanomsg */
130  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
131  return -1);
132 
133  /* initialize and connect subscribe socket */
134  cmcp_node->subscribe_socket = nn_socket(AF_SP, NN_SUB);
135  /* check error set by nanomsg, cleanup publish socket if failed */
136  VSP_CHECK(cmcp_node->subscribe_socket != -1,
137  nn_close(cmcp_node->publish_socket); return -1);
138  /* connect or bind socket */
139  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
140  ret = nn_bind(cmcp_node->subscribe_socket,
141  subscribe_address);
142  } else {
143  ret = nn_connect(cmcp_node->subscribe_socket,
144  subscribe_address);
145  }
146  /* check error set by nanomsg, cleanup both sockets if failed */
147  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
148  nn_close(cmcp_node->subscribe_socket); return -1);
149  /* set receive timeout */
150  ret = nn_setsockopt(cmcp_node->subscribe_socket,
151  NN_SOL_SOCKET, NN_RCVTIMEO,
152  &VSP_CMCP_NODE_TIMEOUT, sizeof(VSP_CMCP_NODE_TIMEOUT));
153  /* check error set by nanomsg, cleanup both sockets if failed */
154  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
155  nn_close(cmcp_node->subscribe_socket); return -1);
156  /* subscribe to broadcast ID */
157  ret = vsp_cmcp_node_subscribe(cmcp_node, VSP_CMCP_BROADCAST_TOPIC_ID);
158  /* check error set by nanomsg, cleanup both sockets if failed */
159  VSP_CHECK(ret == 0, nn_close(cmcp_node->publish_socket);
160  nn_close(cmcp_node->subscribe_socket); return -1);
161 
162  /* set state */
163  cmcp_node->state = VSP_CMCP_NODE_INITIALIZED;
164 
165  /* sockets successfully bound */
166  return 0;
167 }

169 int vsp_cmcp_node_start(vsp_cmcp_node *cmcp_node,
170  void *(*start_routine) (void *), void *param)
171 {
172  int ret;
173 
174  /* check parameter */
175  VSP_ASSERT(cmcp_node != NULL, vsp_error_set_num(EINVAL); return -1);
176 
177  /* check sockets are initialized and thread is not running */
178  VSP_ASSERT(cmcp_node->state == VSP_CMCP_NODE_INITIALIZED,
179  vsp_error_set_num(EINVAL); return -1);
180 
181  /* lock state mutex */
182  pthread_mutex_lock(&cmcp_node->mutex);
183 
184  /* mark thread as starting */
185  cmcp_node->state = VSP_CMCP_NODE_STARTING;
186 
187  /* start reception thread */
188  ret = pthread_create(&cmcp_node->thread, NULL, start_routine, param);
189 
190  /* check for pthread errors */
191  VSP_ASSERT(ret == 0, pthread_mutex_unlock(&cmcp_node->mutex);
192  vsp_error_set_num(ret); return -1);
193 
194  /* wait until thread is running */
195  ret = pthread_cond_wait(&cmcp_node->condition, &cmcp_node->mutex);
196  /* unlock mutex again */
197  pthread_mutex_unlock(&cmcp_node->mutex);
198  /* check for condition waiting errors */
199  VSP_ASSERT(ret == 0, vsp_error_set_num(ret); return -1);
200 
201  /* check thread is running */
202  VSP_ASSERT(cmcp_node->state == VSP_CMCP_NODE_RUNNING,
203  vsp_error_set_num(EINVAL); return -1);
204 
205  /* success */
206  return 0;
207 }

254  /* check for error */
255  VSP_ASSERT(data_buffer != NULL, return -1);

258  ret = vsp_cmcp_message_get_data(cmcp_message, data_buffer);
259  /* check for error */
260  VSP_ASSERT(ret == 0, return -1);
261 
262  /* actually send message to socket */

264  /* check for error */
265  VSP_ASSERT(ret != -1, return -1);

267  /* success */
