138 void vsp_cmcp_node_generate_id(vsp_cmcp_node *cmcp_node)
139 {
140  /* check parameter */
141  VSP_ASSERT(cmcp_node != NULL);
142 
143  /* generate node ID that does not equal broadcast topic ID */
144  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
145  /* server node: ID is even (first bit cleared) */
146  do {
147  cmcp_node->id = (uint16_t) (vsp_random_get() << 1);
148  } while (cmcp_node->id == VSP_CMCP_SERVER_BROADCAST_TOPIC_ID);
149  } else {
150  /* client node: ID is odd (first bit set) */
151  do {
152  cmcp_node->id = (uint16_t) (vsp_random_get() | 1);
153  } while (cmcp_node->id == VSP_CMCP_CLIENT_BROADCAST_TOPIC_ID);
154  }
155 }

165 int vsp_cmcp_node_connect(vsp_cmcp_node *cmcp_node,
166  const char *publish_address, const char *subscribe_address)
167 {
168  int ret;
169 
170  /* check parameters */
171  VSP_ASSERT(cmcp_node != NULL && publish_address != NULL
172  && subscribe_address != NULL);
173  /* check if sockets are not yet initialized */
174  VSP_CHECK(vsp_cmcp_state_get(cmcp_node->state)
175  == VSP_CMCP_NODE_UNINITIALIZED, vsp_error_set_num(EALREADY); return -1);
176 
177  /* initialize and connect publish socket */
178  cmcp_node->publish_socket = nn_socket(AF_SP, NN_PUB);
179  /* vsp_error_num() is set by nn_socket() */
180  VSP_CHECK(cmcp_node->publish_socket != -1, return -1);
181  /* connect or bind socket */
182  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
183  ret = nn_bind(cmcp_node->publish_socket, publish_address);
184  } else {
185  ret = nn_connect(cmcp_node->publish_socket, publish_address);
186  }
187  /* vsp_error_num() is set by nn_bind() or nn_connect() */
188  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
189  return -1);
190 
191  /* initialize and connect subscribe socket */
192  cmcp_node->subscribe_socket = nn_socket(AF_SP, NN_SUB);
193  /* vsp_error_num() is set by nn_socket() */
194  /* cleanup publish socket if failed */
195  VSP_CHECK(cmcp_node->subscribe_socket != -1,
196  nn_close(cmcp_node->publish_socket); return -1);
197  /* connect or bind socket */
198  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
199  ret = nn_bind(cmcp_node->subscribe_socket,
200  subscribe_address);
201  } else {
202  ret = nn_connect(cmcp_node->subscribe_socket,
203  subscribe_address);
204  }
205  /* vsp_error_num() is set by nn_bind() or nn_connect() */
206  /* cleanup both sockets if failed */
207  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
208  nn_close(cmcp_node->subscribe_socket); return -1);
209  /* set receive timeout */
210  ret = nn_setsockopt(cmcp_node->subscribe_socket,
211  NN_SOL_SOCKET, NN_RCVTIMEO, &VSP_CMCP_NODE_HEARTBEAT_TIME, sizeof(int));
212  /* vsp_error_num() is set by nn_setsockopt() */
213  /* cleanup both sockets if failed */
214  VSP_CHECK(ret >= 0, nn_close(cmcp_node->publish_socket);
215  nn_close(cmcp_node->subscribe_socket); return -1);
216 
217  /* set state */
218  vsp_cmcp_state_set(cmcp_node->state, VSP_CMCP_NODE_INITIALIZED);
219 
220  /* subscribe to broadcast ID and node ID */
221  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
222  vsp_cmcp_node_subscribe(cmcp_node, VSP_CMCP_SERVER_BROADCAST_TOPIC_ID);
223  } else {
224  vsp_cmcp_node_subscribe(cmcp_node, VSP_CMCP_CLIENT_BROADCAST_TOPIC_ID);
225  }
226  vsp_cmcp_node_subscribe(cmcp_node, cmcp_node->id);
227 
228  /* sockets successfully bound */
229  return 0;
230 }

232 void vsp_cmcp_node_start(vsp_cmcp_node *cmcp_node)
233 {
234  int ret;
235 
236  /* check parameter */
237  VSP_ASSERT(cmcp_node != NULL);
238 
239  /* check if sockets are initialized and thread is not running */
240  VSP_ASSERT(vsp_cmcp_state_get(cmcp_node->state)
241  == VSP_CMCP_NODE_INITIALIZED);
242 
243  /* mark thread as starting */
244  vsp_cmcp_state_set(cmcp_node->state, VSP_CMCP_NODE_STARTING);
245 
246  /* lock state mutex */
247  vsp_cmcp_state_lock(cmcp_node->state);
248 
249  /* start reception thread */
250  ret = pthread_create(&cmcp_node->thread, NULL,
251  vsp_cmcp_node_run, cmcp_node);
252 
253  /* check for pthread errors */
254  VSP_ASSERT(ret == 0);
255 
256  /* wait until thread is running */
257  ret = vsp_cmcp_state_await_state(cmcp_node->state,
258  VSP_CMCP_NODE_RUNNING, NULL);
259  /* unlock state mutex */
260  vsp_cmcp_state_unlock(cmcp_node->state);
261  /* check for condition waiting errors */
262  VSP_ASSERT(ret == 0);
263 }

289 int vsp_cmcp_node_send_message(int socket, vsp_cmcp_message *cmcp_message)
290 {
291  int ret;
292  int data_length;
293  void *data_buffer;
294 
295  /* get message data length */
296  data_length = vsp_cmcp_message_get_data_length(cmcp_message);
297  /* check for errors */
298  VSP_ASSERT(data_length > 0);
299 
300  /* allocate zero-copy message buffer */
301  data_buffer = nn_allocmsg(data_length, 0);
302  /* check for errors */
303  VSP_ASSERT(data_buffer != NULL);
304 
305  /* write message data to buffer */
306  vsp_cmcp_message_get_data(cmcp_message, data_buffer);
307 
308  /* actually send message to socket */
309  ret = nn_send(socket, &data_buffer, NN_MSG, 0);
310  /* vsp_error_num() is set by nn_send() */
311  VSP_CHECK(ret >= 0, goto error_exit);
312 
313  /* success */
314  return 0;
315 
316  error_exit:
317  /* cleanup */
318  ret = nn_freemsg(data_buffer);
319  VSP_ASSERT(ret == 0);
320  /* vsp_error_num() is already set */
321  return -1;
322 }

397 void *vsp_cmcp_node_run(void *param)
398 {
399  vsp_cmcp_node *cmcp_node;
400  int ret;
401  int data_length;
402  void *message_buffer;
403  vsp_cmcp_message *cmcp_message;
404  uint16_t sender_id;
405 
406  /* check parameter */
407  VSP_ASSERT(param != NULL);
408 
409  /* initialize local variables */
410  cmcp_node = (vsp_cmcp_node*) param;
411  message_buffer = NULL;
412  cmcp_message = NULL;
413 
414  /* check if sockets are initialized and thread was started */
415  VSP_ASSERT(vsp_cmcp_state_get(cmcp_node->state) == VSP_CMCP_NODE_STARTING);
416 
417  /* mark thread as running */
418  vsp_cmcp_state_set(cmcp_node->state, VSP_CMCP_NODE_RUNNING);
419 
420  /* reception loop */
421  while (vsp_cmcp_state_get(cmcp_node->state) == VSP_CMCP_NODE_RUNNING) {
422  /* send heartbeat */
423  vsp_cmcp_node_heartbeat(cmcp_node);
424 
425  /* invoke regular callback function */
426  cmcp_node->regular_callback(cmcp_node->callback_param);
427 
428  /* try to receive message */
429  data_length = nn_recv(cmcp_node->subscribe_socket,
430  &message_buffer, NN_MSG, 0);
431  /* check error: in case of failure just retry */
432  VSP_CHECK(data_length > 0, goto cleanup);
433  /* parse message data */
434  cmcp_message = vsp_cmcp_message_create_parse(data_length,
435  message_buffer);
436  /* check error: in case of failure clean up and retry */
437  VSP_CHECK(cmcp_message != NULL, goto cleanup);
438 
439  /* filter out invalid messages; check if message has valid sender */
440  sender_id = vsp_cmcp_message_get_sender_id(cmcp_message);
441  VSP_CHECK(sender_id != VSP_CMCP_SERVER_BROADCAST_TOPIC_ID
442  && sender_id != VSP_CMCP_CLIENT_BROADCAST_TOPIC_ID, goto cleanup);
443 
444  /* message successfully received; invoke callback function */
445  cmcp_node->message_callback(cmcp_node->callback_param, cmcp_message);
446 
447  cleanup:
448  /* clean up */
449  if (cmcp_message != NULL) {
450  vsp_cmcp_message_free(cmcp_message);
451  cmcp_message = NULL;
452  }
453  if (message_buffer != NULL) {
454  ret = nn_freemsg(message_buffer);
455  VSP_ASSERT(ret == 0);
456  message_buffer = NULL;
457  }
458  continue;
459  }
460 
461  /* check if thread was requested to stop */
462  VSP_ASSERT(vsp_cmcp_state_get(cmcp_node->state) == VSP_CMCP_NODE_STOPPING);
463 
464  /* signal that thread has stopped */
465  vsp_cmcp_state_set(cmcp_node->state, VSP_CMCP_NODE_INITIALIZED);
466 
467  /* success */
468  return (void*) 0;
469 }

471 void vsp_cmcp_node_heartbeat(vsp_cmcp_node *cmcp_node)
472 {
473  double time_now;
474  int ret;
475  uint16_t topic_id, command_id;
476 
477  time_now = vsp_time_real_double();
478  if (time_now < cmcp_node->time_next_heartbeat) {
479  /* not sending heartbeat yet; nothing to fail, hence success */
480  return;
481  }
482 
483  /* update time of next heartbeat */
484  cmcp_node->time_next_heartbeat =
485  time_now + (VSP_CMCP_NODE_HEARTBEAT_TIME / 1000.0);
486 
487  /* send server heartbeat to all clients, client heartbeat to all servers */
488  if (cmcp_node->node_type == VSP_CMCP_NODE_SERVER) {
489  topic_id = VSP_CMCP_CLIENT_BROADCAST_TOPIC_ID;
490  command_id = VSP_CMCP_COMMAND_SERVER_HEARTBEAT;
491  } else {
492  topic_id = VSP_CMCP_SERVER_BROADCAST_TOPIC_ID;
493  command_id = VSP_CMCP_COMMAND_CLIENT_HEARTBEAT;
494  }
495  ret = vsp_cmcp_node_create_send_message(cmcp_node,
496  VSP_CMCP_MESSAGE_TYPE_CONTROL,
497  topic_id, cmcp_node->id, command_id, NULL);
498  /* check for errors */
499  VSP_ASSERT(ret == 0);
500 }
