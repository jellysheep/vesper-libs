20  /* check parameters */
21  VSP_ASSERT(socket != -1, vsp_error_set_num(EINVAL); return -1);
22  VSP_ASSERT(cmcp_message != NULL, vsp_error_set_num(EINVAL); return -1);

24  /* get message data length */
25  data_length = vsp_cmcp_message_get_data_length(cmcp_message);
26  /* check for error */
27  VSP_ASSERT(data_length > 0, return -1);
28 
29  /* allocate zero-copy message buffer */
30  data_buffer = nn_allocmsg(data_length, 0);
31  /* check for error */
32  VSP_ASSERT(data_buffer != NULL, return -1);
33 
34  /* write message data to buffer */
35  ret = vsp_cmcp_message_get_data(cmcp_message, data_buffer);
36  /* check for error */
37  VSP_ASSERT(ret == 0, return -1);
38 
39  /* actually send message to socket */
40  ret = nn_send(socket, &data_buffer, NN_MSG, 0);
41  /* check for error */
42  VSP_ASSERT(ret != -1, return -1);
43 
44  /* success */
45  return 0;
