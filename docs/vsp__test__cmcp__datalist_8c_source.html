125 MU_TEST(vsp_test_cmcp_datalist_test)
126 {
127  vsp_cmcp_datalist *cmcp_datalist2;
128  int ret;
129  int data_length;
130  void *data_pointer;
131  void *data_item_pointer;
132 
133  /* insert data list items */
134  ret = vsp_cmcp_datalist_add_item(global_cmcp_datalist,
135  VSP_TEST_DATALIST_ITEM1_ID,
136  VSP_TEST_DATALIST_ITEM1_LENGTH, VSP_TEST_DATALIST_ITEM1_DATA);
137  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
138 
139  ret = vsp_cmcp_datalist_add_item(global_cmcp_datalist,
140  VSP_TEST_DATALIST_ITEM2_ID,
141  VSP_TEST_DATALIST_ITEM2_LENGTH, VSP_TEST_DATALIST_ITEM2_DATA);
142  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
143 
144  /* get binary data array length */
145  data_length = vsp_cmcp_datalist_get_data_length(global_cmcp_datalist);
146  mu_assert_abort(data_length ==
147  (VSP_TEST_DATALIST_ITEM1_LENGTH + VSP_TEST_DATALIST_ITEM2_LENGTH + 8),
148  vsp_error_str(EINVAL));
149  /* allocate array */
150  data_pointer = malloc(data_length);
151  mu_assert_abort(data_pointer != NULL, vsp_error_str(ENOMEM));
152  /* get binary data */
153  ret = vsp_cmcp_datalist_get_data(global_cmcp_datalist, data_pointer);
154  mu_assert(ret == 0, vsp_error_str(vsp_error_num()));
155 
156  /* construct second data list using binary data array */
157  cmcp_datalist2 = vsp_cmcp_datalist_create_parse(data_length, data_pointer);
158  mu_assert_abort(cmcp_datalist2 != NULL, vsp_error_str(vsp_error_num()));
159 
160  /* insert data ID a second time and check for rejection */
161  ret = vsp_cmcp_datalist_add_item(cmcp_datalist2, VSP_TEST_DATALIST_ITEM1_ID,
162  VSP_TEST_DATALIST_ITEM1_LENGTH, VSP_TEST_DATALIST_ITEM1_DATA);
163  mu_assert(ret != 0, vsp_error_str(EINVAL));
164 
165  /* get back data list item and verify data */
166  data_item_pointer = vsp_cmcp_datalist_get_data_item(cmcp_datalist2,
167  VSP_TEST_DATALIST_ITEM1_ID, VSP_TEST_DATALIST_ITEM1_LENGTH);
168  mu_assert_abort(data_item_pointer != NULL, vsp_error_str(vsp_error_num()));
169  mu_assert(memcmp(data_item_pointer, VSP_TEST_DATALIST_ITEM1_DATA,
170  VSP_TEST_DATALIST_ITEM1_LENGTH) == 0, vsp_error_str(EINVAL));
171 
172  data_item_pointer = vsp_cmcp_datalist_get_data_item(cmcp_datalist2,
173  VSP_TEST_DATALIST_ITEM2_ID, VSP_TEST_DATALIST_ITEM2_LENGTH);
174  mu_assert_abort(data_item_pointer != NULL, vsp_error_str(vsp_error_num()));
175  mu_assert(memcmp(data_item_pointer, VSP_TEST_DATALIST_ITEM2_DATA,
176  VSP_TEST_DATALIST_ITEM2_LENGTH) == 0, vsp_error_str(EINVAL));
177 
178  /* get back item data with wrong length and check for failure */
179  data_item_pointer = vsp_cmcp_datalist_get_data_item(cmcp_datalist2,
180  VSP_TEST_DATALIST_ITEM1_ID, VSP_TEST_DATALIST_ITEM1_LENGTH + 1);
181  mu_assert(data_item_pointer == NULL, vsp_error_str(EINVAL));
182 
183  /* get back item data for unknown item ID and check for failure */
184  data_item_pointer = vsp_cmcp_datalist_get_data_item(cmcp_datalist2,
185  VSP_TEST_DATALIST_ITEM2_ID + 1, VSP_TEST_DATALIST_ITEM2_LENGTH);
186  mu_assert(data_item_pointer == NULL, vsp_error_str(EINVAL));
187 
188  /* deallocation */
189  VSP_FREE(data_pointer);
190  vsp_cmcp_datalist_free(cmcp_datalist2);
191 }
