56 void Vout::threadFunction()
57 {
58  threadRunning = true;
59 
60  std::cout << "[logclass ] loggingThread: started!" << std::endl;
61 
62  while (threadRunning || (messages.empty() == false)) {
63  // thread-safe pop message from queue
64  std::unique_lock<std::mutex> lock(lMutex);
65  // while loop only necessary in case of spurious wakeups
66  while (threadRunning && messages.empty()) {
67  // waiting for messages
68  condVariable.wait(lock);
69  }
70  if (!threadRunning && messages.empty()) {
71  break;
72  }
73 
74  LoggingMessage *loggingMessage = messages.front();
75  messages.pop();
76  lock.unlock();
77 
78  // generate the prefix:
79  std::ostringstream prefix;
80  if (loggingMessage->type == LoggingTypes::client) {
81  prefix << "[client";
82  } else {
83  prefix << "[server";
84  }
85  prefix << "|" << std::setw(5);
86  prefix << loggingMessage->id << "] ";
87  std::string prefixStr = prefix.str();
88 
89  std::string message = loggingMessage->message;
90 
91  // print message and add prefix to newlines:
92  std::string::size_type pos, pos2 = -1;
93  do {
94  pos = pos2 + 1;
95  pos2 = message.find('\n', pos);
96  std::cout << prefixStr;
97  if (pos2 == std::string::npos)
98  std::cout << message.substr(pos);
99  else
100  std::cout << message.substr(pos, pos2-pos);
101  std::cout << '\n';
102  } while (pos2 != std::string::npos);
103  }
