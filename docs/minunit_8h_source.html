87 /* Test runner */
88 #define MU_RUN_TEST(test) MU__SAFE_BLOCK(\
89  if (minunit_real_timer==0 && minunit_real_timer==0) {\
90  minunit_real_timer = mu_timer_real();\
91  minunit_proc_timer = mu_timer_cpu();\
92  }\
93  if (minunit_setup) (*minunit_setup)();\
94  minunit_status = 0;\
95  test();\
96  minunit_run++;\
97  if (minunit_status) {\
98  minunit_fail++;\
99  printf("F");\
100  printf("\n%s\n", minunit_last_message);\
101  }\
102  fflush(stdout);\
103  if (minunit_teardown) (*minunit_teardown)();\
104 )
