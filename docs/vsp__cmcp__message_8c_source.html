72 vsp_cmcp_message *vsp_cmcp_message_create_parse(uint16_t data_length,
73  void *data_pointer)
74 {
75  /* using short int pointer for safe pointer arithmetic */
76  uint16_t *current_data_pointer;
77  vsp_cmcp_message *cmcp_message;
78 
79  /* check parameters */
80  VSP_ASSERT(data_length >= VSP_CMCP_MESSAGE_HEADER_LENGTH,
81  vsp_error_set_num(EINVAL); return NULL);
82  VSP_ASSERT(data_pointer != NULL, vsp_error_set_num(EINVAL); return NULL);
83  /* allocate memory */
84  VSP_ALLOC(cmcp_message, vsp_cmcp_message, return NULL);
85  /* initialize struct data */
86  cmcp_message->type = VSP_CMCP_MESSAGE_TYPE_RECEIVE;
87  current_data_pointer = data_pointer;
88  cmcp_message->topic_id = *current_data_pointer;
89  ++current_data_pointer;
90  cmcp_message->sender_id = *current_data_pointer;
91  ++current_data_pointer;
92  cmcp_message->command_id = *current_data_pointer;
93  ++current_data_pointer;
94  /* parse data list values */
95  cmcp_message->cmcp_datalist = vsp_cmcp_datalist_create_parse(
96  data_length - VSP_CMCP_MESSAGE_HEADER_LENGTH, current_data_pointer);
97  /* in case of failure vsp_error_num() is already set */
98  VSP_ASSERT(cmcp_message->cmcp_datalist != NULL, return NULL);
99 
100  /* return struct pointer */
101  return cmcp_message;
102 }
