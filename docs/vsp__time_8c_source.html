28 void vsp_time_real_timespec(struct timespec *time)
29 {
30 #if defined(_WIN32)
31  FILETIME tm;
32  ULONGLONG t;
33  /* check parameter */
34  VSP_ASSERT(time != NULL);
35  #if defined(NTDDI_WIN8) && NTDDI_VERSION >= NTDDI_WIN8
36  /* Windows 8, Windows Server 2012 and later */
37  GetSystemTimePreciseAsFileTime(&tm);
38  #else
39  /* Windows 2000 and later */
40  GetSystemTimeAsFileTime(&tm);
41  #endif
42  t = ((ULONGLONG)tm.dwHighDateTime << 32) | (ULONGLONG)tm.dwLowDateTime;
43  t -= VSP_TIME_EPOCH_FILETIME;
44  time->tv_sec = t / 10000000;
45  time->tv_nsec = (t % 10000000) * 100;
46  return;
47 
48 #elif defined(_POSIX_VERSION)
49  struct timeval tv;
50  /* check parameter */
51  VSP_ASSERT(time != NULL);
52  /* POSIX */
53  #if defined(_POSIX_C_SOURCE) && _POSIX_C_SOURCE >= 200112L \
54  && defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0) \
55  && defined(CLOCK_REALTIME)
56  {
57  /* AIX, BSD, HP-UX, Linux, POSIX */
58  const clockid_t id = CLOCK_REALTIME;
59  if (clock_gettime(id, time) != -1) {
60  return;
61  }
62  /* else fall through */
63  }
64  #endif
65  /* AIX, BSD, Cygwin, HP-UX, Linux, OSX, POSIX, Solaris */
66  gettimeofday(&tv, NULL);
67  time->tv_sec = tv.tv_sec;
68  time->tv_nsec = tv.tv_usec * 1000;
69  return;
70 #else
71  #error "Unable to define timers for an unknown OS."
72 #endif /* defined(_WIN32) */
73 }

75 void vsp_time_real_timespec_from_now(struct timespec *time,
76  unsigned int milliseconds)
77 {
78  vsp_time_real_timespec(time);
79  /* add fractional seconds */
80  time->tv_nsec += (milliseconds % 1000) * 1000000l;
81  /* remove overflow */
82  if (time->tv_nsec > 1000000000l) {
83  time->tv_nsec -= 1000000000l;
84  time->tv_sec += 1;
85  }
86  /* add whole seconds */
87  time->tv_sec += milliseconds / 1000;
88 }

115 double vsp_time_cpu_double()
116 {
117 #if defined(_WIN32)
118  /* Windows */
119  FILETIME createTime;
120  FILETIME exitTime;
121  FILETIME kernelTime;
122  FILETIME userTime;
123  if (GetProcessTimes(GetCurrentProcess(),
124  &createTime, &exitTime, &kernelTime, &userTime) != -1) {
125  SYSTEMTIME userSystemTime;
126  if (FileTimeToSystemTime(&userTime, &userSystemTime) != -1) {
127  return (double)userSystemTime.wHour * 3600.0 +
128  (double)userSystemTime.wMinute * 60.0 +
129  (double)userSystemTime.wSecond +
130  (double)userSystemTime.wMilliseconds / 1000.0;
131  }
132  }
133 
134 #elif defined(__unix__) || defined(__unix) || defined(unix) \
135  || (defined(__APPLE__) && defined(__MACH__))
136  /* AIX, BSD, Cygwin, HP-UX, Linux, OSX, and Solaris */
137 
138  #if defined(_POSIX_C_SOURCE) && _POSIX_C_SOURCE >= 200112L \
139  && defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
140  /* prefer high-res POSIX timers, when available */
141  {
142  clockid_t id;
143  struct timespec ts;
144  #if _POSIX_CPUTIME > 0
145  /* clock ids vary by OS. Query the id, if possible */
146  if (clock_getcpuclockid(0, &id) == -1)
147  #endif /* _POSIX_CPUTIME > 0 */
148  {
149  #if defined(CLOCK_PROCESS_CPUTIME_ID)
150  /* use known clock id for AIX, Linux, or Solaris */
151  id = CLOCK_PROCESS_CPUTIME_ID;
152  #elif defined(CLOCK_VIRTUAL)
153  /* use known clock id for BSD or HP-UX */
154  id = CLOCK_VIRTUAL;
155  #else
156  id = (clockid_t)-1;
157  #endif /* defined(CLOCK_PROCESS_CPUTIME_ID) */
158  }
159  if (id != (clockid_t)-1 && clock_gettime(id, &ts) != -1) {
160  return (double)ts.tv_sec +
161  (double)ts.tv_nsec / 1000000000.0;
162  }
163  }
164  #endif /* defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0) */
165 
166  #if defined(RUSAGE_SELF)
167  {
168  struct rusage rusage;
169  if (getrusage(RUSAGE_SELF, &rusage) != -1) {
170  return (double)rusage.ru_utime.tv_sec +
171  (double)rusage.ru_utime.tv_usec / 1000000.0;
172  }
173  }
174  #endif /* defined(RUSAGE_SELF) */
175 
176  #if defined(_SC_CLK_TCK)
177  {
178  const double ticks = (double)sysconf(_SC_CLK_TCK);
179  struct tms tms;
180  if (times(&tms) != (clock_t)-1) {
181  return (double)tms.tms_utime / ticks;
182  }
183  }
184  #endif /* defined(_SC_CLK_TCK) */
185 
186  #if defined(CLOCKS_PER_SEC)
187  {
188  clock_t cl = clock();
189  if (cl != (clock_t)-1) {
190  return (double)cl / (double)CLOCKS_PER_SEC;
191  }
192  }
193  #endif /* defined(CLOCKS_PER_SEC) */
194 
195 #endif /* defined(_WIN32) */
196 
197  return -1; /* failed */
198 }
