45 double vsp_time_real(void)
46 {
47 #if defined(_WIN32)
48  FILETIME tm;
49  ULONGLONG t;
50  #if defined(NTDDI_WIN8) && NTDDI_VERSION >= NTDDI_WIN8
51  /* Windows 8, Windows Server 2012 and later */
52  GetSystemTimePreciseAsFileTime(&tm);
53  #else
54  /* Windows 2000 and later */
55  GetSystemTimeAsFileTime(&tm);
56  #endif
57  t = ((ULONGLONG)tm.dwHighDateTime << 32) | (ULONGLONG)tm.dwLowDateTime;
58  return (double)t / 10000000.0;
59 
60 #elif (defined(__hpux) || defined(hpux)) || ((defined(__sun__) || \
61  defined(__sun) || defined(sun)) && (defined(__SVR4) || defined(__svr4__)))
62  /* HP-UX, Solaris */
63  return (double)gethrtime() / 1000000000.0;
64 
65 #elif defined(__MACH__) && defined(__APPLE__)
66  /* OSX */
67  static double timeConvert = 0.0;
68  if (timeConvert == 0.0) {
69  mach_timebase_info_data_t timeBase;
70  (void)mach_timebase_info(&timeBase);
71  timeConvert = (double)timeBase.numer /
72  (double)timeBase.denom /
73  1000000000.0;
74  }
75  return (double)mach_absolute_time() * timeConvert;
76 
77 #elif defined(_POSIX_VERSION)
78  struct timeval tm;
79  /* POSIX */
80  #if defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
81  {
82  struct timespec ts;
83  #if defined(CLOCK_MONOTONIC_PRECISE)
84  /* BSD */
85  const clockid_t id = CLOCK_MONOTONIC_PRECISE;
86  #elif defined(CLOCK_MONOTONIC_RAW)
87  /* Linux */
88  const clockid_t id = CLOCK_MONOTONIC_RAW;
89  #elif defined(CLOCK_HIGHRES)
90  /* Solaris */
91  const clockid_t id = CLOCK_HIGHRES;
92  #elif defined(CLOCK_MONOTONIC)
93  /* AIX, BSD, Linux, POSIX, Solaris */
94  const clockid_t id = CLOCK_MONOTONIC;
95  #elif defined(CLOCK_REALTIME)
96  /* AIX, BSD, HP-UX, Linux, POSIX */
97  const clockid_t id = CLOCK_REALTIME;
98  #else
99  const clockid_t id = (clockid_t)-1; /* unknown */
100  #endif /* defined(CLOCK_MONOTONIC_PRECISE) */
101  if (id != (clockid_t)-1 && clock_gettime(id, &ts) != -1) {
102  return (double)ts.tv_sec +
103  (double)ts.tv_nsec / 1000000000.0;
104  }
105  /* fall through */
106  }
107  #endif /* defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0) */
108 
109  /* AIX, BSD, Cygwin, HP-UX, Linux, OSX, POSIX, Solaris */
110  gettimeofday(&tm, NULL);
111  return (double)tm.tv_sec + (double)tm.tv_usec / 1000000.0;
112 #else
113  return -1.0; /* failed */
114 #endif /* defined(_WIN32) */
115 }

117 double vsp_time_cpu(void)
118 {
119 #if defined(_WIN32)
120  /* Windows */
121  FILETIME createTime;
122  FILETIME exitTime;
123  FILETIME kernelTime;
124  FILETIME userTime;
125  if (GetProcessTimes(GetCurrentProcess(),
126  &createTime, &exitTime, &kernelTime, &userTime) != -1) {
127  SYSTEMTIME userSystemTime;
128  if (FileTimeToSystemTime(&userTime, &userSystemTime) != -1) {
129  return (double)userSystemTime.wHour * 3600.0 +
130  (double)userSystemTime.wMinute * 60.0 +
131  (double)userSystemTime.wSecond +
132  (double)userSystemTime.wMilliseconds / 1000.0;
133  }
134  }
135 
136 #elif defined(__unix__) || defined(__unix) || defined(unix) \
137  || (defined(__APPLE__) && defined(__MACH__))
138  /* AIX, BSD, Cygwin, HP-UX, Linux, OSX, and Solaris */
139 
140  #if defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0)
141  /* prefer high-res POSIX timers, when available */
142  {
143  clockid_t id;
144  struct timespec ts;
145  #if _POSIX_CPUTIME > 0
146  /* clock ids vary by OS. Query the id, if possible */
147  if (clock_getcpuclockid(0, &id) == -1)
148  #endif /* _POSIX_CPUTIME > 0 */
149  {
150  #if defined(CLOCK_PROCESS_CPUTIME_ID)
151  /* use known clock id for AIX, Linux, or Solaris */
152  id = CLOCK_PROCESS_CPUTIME_ID;
153  #elif defined(CLOCK_VIRTUAL)
154  /* use known clock id for BSD or HP-UX */
155  id = CLOCK_VIRTUAL;
156  #else
157  id = (clockid_t)-1;
158  #endif /* defined(CLOCK_PROCESS_CPUTIME_ID) */
159  }
160  if (id != (clockid_t)-1 && clock_gettime(id, &ts) != -1) {
161  return (double)ts.tv_sec +
162  (double)ts.tv_nsec / 1000000000.0;
163  }
164  }
165  #endif /* defined(_POSIX_TIMERS) && (_POSIX_TIMERS > 0) */
166 
167  #if defined(RUSAGE_SELF)
168  {
169  struct rusage rusage;
170  if (getrusage(RUSAGE_SELF, &rusage) != -1) {
171  return (double)rusage.ru_utime.tv_sec +
172  (double)rusage.ru_utime.tv_usec / 1000000.0;
173  }
174  }
175  #endif /* defined(RUSAGE_SELF) */
176 
177  #if defined(_SC_CLK_TCK)
178  {
179  const double ticks = (double)sysconf(_SC_CLK_TCK);
180  struct tms tms;
181  if (times(&tms) != (clock_t)-1) {
182  return (double)tms.tms_utime / ticks;
183  }
184  }
185  #endif /* defined(_SC_CLK_TCK) */
186 
187  #if defined(CLOCKS_PER_SEC)
188  {
189  clock_t cl = clock();
190  if (cl != (clock_t)-1) {
191  return (double)cl / (double)CLOCKS_PER_SEC;
192  }
193  }
194  #endif /* defined(CLOCKS_PER_SEC) */
195 
196 #endif /* defined(_WIN32) */
197 
198  return -1; /* failed */
199 }
