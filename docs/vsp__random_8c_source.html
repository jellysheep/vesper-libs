28 void vsp_random_initialize(void)
29 {
30  uint64_t seed;
31  double realtime;
32  uint64_t realtime_int;
33  uint64_t pid;
34 
35  if (vsp_random_value != 0) {
36  return;
37  }
38 
39  /* get random data from current time and process id */
40  realtime = vsp_time_real();
41  realtime_int = *(uint64_t*) &realtime;
42  #if defined(_WIN32)
43  pid = (uint64_t) GetCurrentProcessId();
44  #else
45  pid = (uint64_t) getpid();
46  #endif
47  /* mix random data into seed */
48  seed = realtime_int;
49  seed ^= pid;
50  seed ^= (uint32_t) (seed >> 32);
51  /* store seed */
52  vsp_random_value = (uint32_t) seed;
53 }

55 uint32_t vsp_random_get(void)
56 {
57  uint32_t bit;
58  /* initialize generator seed */
59  vsp_random_initialize();
60  /* generate next linear feedback shift value bit */
61  /* uses xor from bits 32, 22, 2, 1 */
62  bit = ((vsp_random_value >> 0) ^ (vsp_random_value >> 10)
63  ^ (vsp_random_value >> 30) ^ (vsp_random_value >> 31)) & 1;
64  /* update and return value */
65  vsp_random_value = (vsp_random_value >> 1) | (bit << 31);
66  return vsp_random_value;
67 }
