54 int vsp_cmcp_connect(vsp_cmcp_connector *net_conn,
55  const char *publish_address, const char *subscribe_address)
56 {
57  int ret;
58 
59  /* check parameters */
60  VSP_ASSERT(net_conn != NULL && publish_address != NULL
61  && subscribe_address != NULL, vsp_error_set_num(EINVAL); return -1);
62  /* check sockets not yet initialized */
63  VSP_ASSERT(net_conn->state == VSP_CMCP_UNINITIALIZED,
64  vsp_error_set_num(EALREADY); return -1);
65 
66  /* initialize sockets */
67  net_conn->publish_socket = nn_socket(AF_SP, NN_PUB);
68  net_conn->subscribe_socket = nn_socket(AF_SP, NN_SUB);
69 
70  /* connect sockets */
71  ret = nn_connect(net_conn->publish_socket, publish_address);
72  /* check error set by nanomsg */
73  VSP_ASSERT(ret >= 0, return -1);
74 
75  ret = nn_connect(net_conn->subscribe_socket, subscribe_address);
76  /* check error set by nanomsg */
77  VSP_ASSERT(ret >= 0, return -1);
78 
79  /* set state */
80  net_conn->state = VSP_CMCP_INITIALIZED;
81  /* sockets successfully connected */
82  return 0;
83 }

96  /* disconnect sockets */

102  /* check error set by nanomsg */
103  VSP_ASSERT(ret == 0, return -1);
104

107  net_conn->subscribe_socket = -1;
108
