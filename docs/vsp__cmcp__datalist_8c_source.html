55 int vsp_cmcp_datalist_add_item(vsp_cmcp_datalist *cmcp_datalist,
56  int data_id, int data_length, void *data_pointer)
57 {
58  /* check parameter */
59  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return -1);
60 
61  /* check data list item count */
62  VSP_ASSERT(cmcp_datalist->data_item_count < VSP_CMCP_DATALIST_MAX_ITEMS,
63  vsp_error_set_num(ENOMEM); return -1);
64 
65  /* check data list item was not added yet */
66  VSP_CHECK(_vsp_cmcp_datalist_find_item(cmcp_datalist, data_id) == -1,
67  vsp_error_set_num(EALREADY); return -1);
68 
69  /* add data list item */
70  cmcp_datalist->data_ids[cmcp_datalist->data_item_count] = data_id;
71  cmcp_datalist->data_lengths[cmcp_datalist->data_item_count] = data_length;
72  cmcp_datalist->data_pointers[cmcp_datalist->data_item_count] = data_pointer;
73  ++cmcp_datalist->data_item_count;
74 
75  /* success */
76  return 0;
77 }

79 void *vsp_cmcp_datalist_get_data(vsp_cmcp_datalist *cmcp_datalist,
80  int data_id, int data_length)
81 {
82  int index;
83 
84  /* check parameter */
85  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return NULL);
86 
87  /* search for data ID */
88  index = _vsp_cmcp_datalist_find_item(cmcp_datalist, data_id);
89 
90  /* check data list item index */
91  VSP_CHECK(index != -1, vsp_error_set_num(EINVAL); return NULL);
92 
93  /* check data list item length */
94  VSP_CHECK(cmcp_datalist->data_lengths[index] == data_length,
95  vsp_error_set_num(EINVAL); return NULL);
96 
97  /* return data pointer */
98  return cmcp_datalist->data_pointers[index];
99 }

101 int _vsp_cmcp_datalist_find_item(vsp_cmcp_datalist *cmcp_datalist, int data_id)
102 {
103  int index;
104 
105  /* check parameter */
106  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return -1);
107 
108  for (index = 0; index < cmcp_datalist->data_item_count; ++index) {
109  if (cmcp_datalist->data_ids[index] == data_id) {
110  /* data ID found */
111  return index;
112  }
113  }
114 
115  /* data ID not found */
116  return -1;
117 }
