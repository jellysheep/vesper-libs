55 vsp_cmcp_datalist *vsp_cmcp_datalist_create_parse(uint16_t data_length,
56  void *data_pointer)
57 {
58  uint16_t data_item_id;
59  uint16_t data_item_length;
60  void *data_item_pointer;
61  /* using byte pointer for safe pointer arithmetic */
62  uint8_t *current_data_pointer;
63 
64  vsp_cmcp_datalist *cmcp_datalist;
65  /* allocate memory */
66  VSP_ALLOC(cmcp_datalist, vsp_cmcp_datalist, return NULL);
67  /* initialize struct data: set number of list items to zero */
68  cmcp_datalist->data_item_count = 0;
69  /* add data list items */
70  current_data_pointer = data_pointer;
71  while (data_length >= 4) {
72  data_item_id = *(uint16_t*) current_data_pointer;
73  current_data_pointer += 2;
74  data_length -= 2;
75 
76  data_item_length = *(uint16_t*) current_data_pointer;
77  current_data_pointer += 2;
78  data_length -= 2;
79 
80  data_item_pointer = current_data_pointer;
81  current_data_pointer += data_item_length;
82  data_length -= data_item_length;
83 
84  /* add data list item; failures are silently ignored */
85  vsp_cmcp_datalist_add_item(cmcp_datalist, data_item_id,
86  data_item_length, data_item_pointer);
87  }
88  /* return struct pointer */
89  return cmcp_datalist;
90 }

92 int vsp_cmcp_datalist_get_data(vsp_cmcp_datalist *cmcp_datalist,
93  uint16_t *data_length, void **data_pointer)
94 {
95  /* max. 64 KiB of data allowed */
96  uint16_t _data_length;
97  void *_data_pointer;
98  uint16_t index;
99  uint16_t data_item_length;
100  /* using byte pointer for safe pointer arithmetic */
101  uint8_t *current_data_pointer;
102 
103  /* check parameter */
104  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return -1);
105 
106  /* calculate data length */
107  for (index = 0; index < cmcp_datalist->data_item_count; ++index) {
108  /* 2 bytes per data item id and length */
109  _data_length += 4;
110  /* additional bytes per data item */
111  _data_length += cmcp_datalist->data_item_lengths[index];
112  }
113 
114  /* allocate memory */
115  VSP_ALLOC_N(_data_pointer, _data_length, return -1);
116 
117  /* store data item values */
118  current_data_pointer = _data_pointer;
119  for (index = 0; index < cmcp_datalist->data_item_count; ++index) {
120  *(uint16_t*) current_data_pointer =
121  cmcp_datalist->data_item_ids[index];
122  current_data_pointer += 2;
123 
124  data_item_length = cmcp_datalist->data_item_lengths[index];
125  *(uint16_t*) current_data_pointer = data_item_length;
126  current_data_pointer += 2;
127 
128  if (data_item_length > 0) {
129  memcpy(current_data_pointer,
130  cmcp_datalist->data_item_pointers[index], data_item_length);
131  current_data_pointer += data_item_length;
132  }
133  }
134 
135  /* store data array and length in input parameters */
136  *data_length = _data_length;
137  *data_pointer = _data_pointer;
138 
139  /* success */
140  return 0;
141 }

143 int vsp_cmcp_datalist_add_item(vsp_cmcp_datalist *cmcp_datalist,
144  uint16_t data_item_id, uint16_t data_item_length, void *data_item_pointer)
145 {
146  /* check parameter */
147  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return -1);
148 
149  /* check data list item count */
150  VSP_ASSERT(cmcp_datalist->data_item_count < VSP_CMCP_DATALIST_MAX_ITEMS,
151  vsp_error_set_num(ENOMEM); return -1);
152 
153  /* check data list item was not added yet */
154  VSP_CHECK(_vsp_cmcp_datalist_find_item(cmcp_datalist, data_item_id) == -1,
155  vsp_error_set_num(EALREADY); return -1);
156 
157  /* add data list item */
158  cmcp_datalist->data_item_ids[cmcp_datalist->data_item_count] = data_item_id;
159  cmcp_datalist->data_item_lengths[cmcp_datalist->data_item_count] =
160  data_item_length;
161  cmcp_datalist->data_item_pointers[cmcp_datalist->data_item_count] =
162  data_item_pointer;
163  ++cmcp_datalist->data_item_count;
164 
165  /* success */
166  return 0;
167 }

172  int index;

174  /* check parameter */

176 
177  /* search for data ID */
178  index = _vsp_cmcp_datalist_find_item(cmcp_datalist, data_item_id);
179 
180  /* check data list item index */
181  VSP_CHECK(index != -1, vsp_error_set_num(EINVAL); return NULL);
182 
183  /* check data list item length */
184  VSP_CHECK(cmcp_datalist->data_item_lengths[index] == data_item_length,
185  vsp_error_set_num(EINVAL); return NULL);
186 
187  /* return data pointer */
188  return cmcp_datalist->data_item_pointers[index];
189 }

191 int _vsp_cmcp_datalist_find_item(vsp_cmcp_datalist *cmcp_datalist,
192  uint16_t data_item_id)
193 {
194  int index;
195 
196  /* check parameter */
197  VSP_ASSERT(cmcp_datalist != NULL, vsp_error_set_num(EINVAL); return -1);
198 
199  for (index = 0; index < cmcp_datalist->data_item_count; ++index) {
200  if (cmcp_datalist->data_item_ids[index] == data_item_id) {
201  /* data ID found */
202  return index;
203  }
204  }
205 
206  /* data ID not found */
207  return -1;
208 }
