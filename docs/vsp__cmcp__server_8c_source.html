73 vsp_cmcp_server *vsp_cmcp_server_create(void)
74 {
75  vsp_cmcp_server *cmcp_server;
76  /* allocate memory */
77  VSP_ALLOC(cmcp_server, vsp_cmcp_server, return NULL);
78  /* initialize struct data */
79  cmcp_server->state = VSP_CMCP_SERVER_UNINITIALIZED;
80  /* generate server ID that is even (first bit cleared)
81  * but does not equal broadcast topic ID */
82  do {
83  cmcp_server->id = (uint16_t) (vsp_random_get() << 1);
84  } while (cmcp_server->id == VSP_CMCP_BROADCAST_TOPIC_ID);
85  cmcp_server->publish_socket = -1;
86  cmcp_server->subscribe_socket = -1;
87  cmcp_server->time_next_heartbeat = vsp_time_real();
88  pthread_mutex_init(&cmcp_server->mutex, NULL);
89  pthread_cond_init(&cmcp_server->condition, NULL);
90  /* return struct pointer */
91  return cmcp_server;
92 }

94 int vsp_cmcp_server_free(vsp_cmcp_server *cmcp_server)
95 {
96  int ret;
97  int success;
98 
99  success = 0;
100 
101  /* check parameter */
102  VSP_ASSERT(cmcp_server != NULL, vsp_error_set_num(EINVAL); return -1);
103 
104  if (cmcp_server->state > VSP_CMCP_SERVER_INITIALIZED) {
105  /* worker thread is running, stop it */
106  ret = vsp_cmcp_server_stop(cmcp_server);
107  /* check for error */
108  VSP_ASSERT(ret == 0, success = -1);
109  }
110 
111  if (cmcp_server->state > VSP_CMCP_SERVER_UNINITIALIZED) {
112  /* close publish socket */
113  ret = nn_close(cmcp_server->publish_socket);
114  /* check error set by nanomsg */
115  VSP_ASSERT(ret == 0, success = -1);
116 
117  /* close subscribe socket */
118  ret = nn_close(cmcp_server->subscribe_socket);
119  /* check error set by nanomsg */
120  VSP_ASSERT(ret == 0, success = -1);
121  }
122  pthread_mutex_destroy(&cmcp_server->mutex);
123  pthread_cond_destroy(&cmcp_server->condition);
124  /* free memory */
125  VSP_FREE(cmcp_server);
126  return success;

166 int vsp_cmcp_server_start(vsp_cmcp_server *cmcp_server)
167 {
168  int ret;
169 
170  /* check parameter */
171  VSP_ASSERT(cmcp_server != NULL, vsp_error_set_num(EINVAL); return -1);
172 
173  /* check sockets are initialized and thread is not running */
174  VSP_ASSERT(cmcp_server->state == VSP_CMCP_SERVER_INITIALIZED,
175  vsp_error_set_num(EINVAL); return -1);
176 
177  /* lock state mutex */
178  pthread_mutex_lock(&cmcp_server->mutex);
179 
180  /* mark thread as starting */
181  cmcp_server->state = VSP_CMCP_SERVER_STARTING;
182 
183  /* start reception thread */
184  ret = pthread_create(&cmcp_server->thread, NULL, vsp_cmcp_server_run,
185  (void*) cmcp_server);
186 
187  /* check for pthread errors */
188  VSP_ASSERT(ret == 0, pthread_mutex_unlock(&cmcp_server->mutex);
189  vsp_error_set_num(ret); return -1);
190 
191  /* wait until thread is running */
192  ret = pthread_cond_wait(&cmcp_server->condition, &cmcp_server->mutex);
193  /* unlock mutex again */
194  pthread_mutex_unlock(&cmcp_server->mutex);
195  /* check for condition waiting errors */
196  VSP_ASSERT(ret == 0, vsp_error_set_num(ret); return -1);
197 
198  /* check thread is running */
199  VSP_ASSERT(cmcp_server->state == VSP_CMCP_SERVER_RUNNING,
200  vsp_error_set_num(EINVAL); return -1);
201 
202  /* success */
203  return 0;
204 }

234 void *vsp_cmcp_server_run(void *param)
235 {
236  vsp_cmcp_server *cmcp_server;
237  int ret;
238 
239  /* check parameter */
240  VSP_ASSERT(param != NULL, vsp_error_set_num(EINVAL); return (void*) -1);
241 
242  cmcp_server = (vsp_cmcp_server*) param;
243 
244  /* check sockets are initialized and thread was started */
245  VSP_ASSERT(cmcp_server->state == VSP_CMCP_SERVER_STARTING,
246  vsp_error_set_num(EINVAL); return (void*) -1);
247 
248  /* lock mutex */
249  pthread_mutex_lock(&cmcp_server->mutex);
250  /* mark thread as running */
251  cmcp_server->state = VSP_CMCP_SERVER_RUNNING;
252  /* signal waiting thread */
253  pthread_cond_signal(&cmcp_server->condition);
254  /* unlock mutex */
255  pthread_mutex_unlock(&cmcp_server->mutex);
256 
257  /* reception loop */
258  while (cmcp_server->state == VSP_CMCP_SERVER_RUNNING) {
259  ret = vsp_cmcp_server_heartbeat(cmcp_server);
260  /* check error */
261  VSP_ASSERT(ret == 0, return (void*) -1);
262  }
263 
264  /* check thread was requested to stop */
265  VSP_ASSERT(cmcp_server->state == VSP_CMCP_SERVER_STOPPING,
266  vsp_error_set_num(EINTR); return (void*) -1);
267 
268  /* signal that thread has stopped */
269  cmcp_server->state = VSP_CMCP_SERVER_INITIALIZED;
270 
271  /* success */
272  return (void*) 0;
273 }

275 int vsp_cmcp_server_heartbeat(vsp_cmcp_server *cmcp_server)
276 {
277  double time_now;
278  int ret;
279 
280  /* check parameter */
281  VSP_ASSERT(cmcp_server != NULL, vsp_error_set_num(EINVAL); return -1);
282 
283  time_now = vsp_time_real();
284  if (time_now < cmcp_server->time_next_heartbeat) {
285  /* not sending heartbeat yet; nothing to fail, hence success */
286  return 0;
287  }
288 
289  /* update time of next heartbeat */
290  cmcp_server->time_next_heartbeat =
291  time_now + VSP_CMCP_SERVER_HEARTBEAT_TIME;
292 
293  /* send heartbeat */
294  ret = vsp_cmcp_common_create_send_message(cmcp_server->publish_socket,
295  0, cmcp_server->id, VSP_CMCP_COMMAND_HEARTBEAT, NULL);
296  /* check for error */
297  VSP_ASSERT(ret == 0, return -1);
298 
299  return 0;
