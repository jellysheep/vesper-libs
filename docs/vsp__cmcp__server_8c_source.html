96 void *vsp_cmcp_server_run(void *param)
97 {
98  vsp_cmcp_server *cmcp_server;
99  int ret;
100 
101  /* check parameter */
102  VSP_ASSERT(param != NULL, vsp_error_set_num(EINVAL); return (void*) -1);
103 
104  cmcp_server = (vsp_cmcp_server*) param;
105 
106  /* check sockets are initialized and thread was started */
107  VSP_ASSERT(cmcp_server->cmcp_node->state == VSP_CMCP_NODE_STARTING,
108  vsp_error_set_num(EINVAL); return (void*) -1);
109 
110  /* lock mutex */
111  pthread_mutex_lock(&cmcp_server->cmcp_node->mutex);
112  /* mark thread as running */
113  cmcp_server->cmcp_node->state = VSP_CMCP_NODE_RUNNING;
114  /* signal waiting thread */
115  pthread_cond_signal(&cmcp_server->cmcp_node->condition);
116  /* unlock mutex */
117  pthread_mutex_unlock(&cmcp_server->cmcp_node->mutex);
118 
119  /* reception loop */
120  while (cmcp_server->cmcp_node->state == VSP_CMCP_NODE_RUNNING) {
121  ret = vsp_cmcp_server_heartbeat(cmcp_server);
122  /* check error */
123  VSP_ASSERT(ret == 0, return (void*) -1);
124  }
125 
126  /* check thread was requested to stop */
127  VSP_ASSERT(cmcp_server->cmcp_node->state == VSP_CMCP_NODE_STOPPING,
128  vsp_error_set_num(EINTR); return (void*) -1);
129 
130  /* signal that thread has stopped */
131  cmcp_server->cmcp_node->state = VSP_CMCP_NODE_INITIALIZED;
132 
133  /* success */
134  return (void*) 0;
135 }

137 int vsp_cmcp_server_heartbeat(vsp_cmcp_server *cmcp_server)
138 {
139  double time_now;
140  int ret;
141 
142  /* check parameter */
143  VSP_ASSERT(cmcp_server != NULL, vsp_error_set_num(EINVAL); return -1);
144 
145  time_now = vsp_time_real();
146  if (time_now < cmcp_server->time_next_heartbeat) {
147  /* not sending heartbeat yet; nothing to fail, hence success */
148  return 0;
149  }
150 
151  /* update time of next heartbeat */
152  cmcp_server->time_next_heartbeat =
153  time_now + (VSP_CMCP_SERVER_HEARTBEAT_TIME / 1000.0);
154 
155  /* send heartbeat */
156  ret = vsp_cmcp_node_create_send_message(cmcp_server->cmcp_node,
157  0, cmcp_server->cmcp_node->id, VSP_CMCP_COMMAND_HEARTBEAT, NULL);
158  /* check for error */
159  VSP_ASSERT(ret == 0, return -1);
160 
161  return 0;
162 }
